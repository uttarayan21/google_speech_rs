Represents Python’s OverflowError exception.

§Example: Raising OverflowError from Rust

This exception can be sent to Python code by converting it into a PyErr, where Python code can then catch it.

use pyo3::prelude::*;
use pyo3::exceptions::PyOverflowError;

#[pyfunction]
fn always_throws() -> PyResult<()> {
    let message = "I'm OverflowError, and I was raised from Rust.";
    Err(PyOverflowError::new_err(message))
}

Python code:

from my_module import always_throws

try:
   always_throws()
except OverflowError as e:
   print(f"Caught an exception: {e}")

§Example: Catching OverflowError in Rust

use pyo3::prelude::*;
use pyo3::exceptions::PyOverflowError;
use pyo3::ffi::c_str;

Python::with_gil(|py| {
    let result: PyResult<()> = py.run(c_str!("raise OverflowError"), None, None);

    let error_type = match result {
        Ok(_) => "Not an error",
        Err(error) if error.is_instance_of::<PyOverflowError>(py) => "OverflowError",
        Err(_) => "Some other error",
    };

    assert_eq!(error_type, "OverflowError");
});
